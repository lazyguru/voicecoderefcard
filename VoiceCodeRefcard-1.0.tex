\section{General Text Commands}
\begin{description}
\item [tarp] Sends the tab key
\item [skoosh] Inserts a space
\item [shock] Sends the enter key
\item [randall] Sends the escape key
\end{description}

\section{Editing Commands}
\begin{description}
\item [scratchy] Attempts a "smart" undo
\item [dizzle] Undo
\item [rizzle] Redo
\item [snatch] cut
\item [stoosh] copy
\item [spark] paste
\item [junk] delete
\item [olly] select all
\item [shackle] select entire line
\item [snipline] delete line
\item [snipper] delete to end of line (from cursor)
\item [snipple] delete to beginning of line (from cursor)
\item [jolt] duplicate line
\item [marco] find
\item [freshly] reload/refresh (depends on context)
\end{description}

\section{Mouse}
\begin{description}
\item [chiff] Single left click
\item [duke] Double-click
\item [chipper] Right click
\item [chiffpark] Click then paste
\item [dookoosh] Double-click then copy
\item [shoff skoosh] Shortcat
\end{description}

\section{Movement}
\begin{description}
\item [jeep] Up
\item [doom] Down
\item [crimp] Left
\item [chris] Right
\item [prome] Home
\item [prend] End
\item [skunkrim] Move the cursor by word to the left
\item [skunkrish] Move the cursor by word to the right
\item [fox appname] Open named application
\item [swick] Switch apps
\item [gibby/shibby] Switch window in app
\end{description}

\section{Text Manipulation}
\begin{description}

\item [dollcram] \$camelCase
\item [spine] Spinal-case
\end{description}

\section{File/Window Operations}
\begin{description}
\item [sage] Save file
\item [totch] Close window/tab
\item [talky] Open new tab
\end{description}

\section{Symbols}
\begin{description}
\item [shalls] \textbackslash
\item [questo] ?
\item [smaqual] =
\item [coif] Opening double-quote
\item [posh] Opening single-quote
\item [brax] Inserts brackets and positions cursor between them
\end{description}

\section{Sublime Text}
\begin{description}
\item [spring] Go to line number
\item [sprinkler] Go to line number then position cursor at end of line.
\item [sprinkoon] Go to line number then insert a new line below.
\item [spackle] Go to line number then select entire line.
\item [bracken] Expand selection to quotes, parens, braces, or brackets
\item [selrang] Selects text in a line range: selrang ten twenty
\item [seltil] Selects text from current position through spoken line number: seltil five five.
\item [trundle] Toggle comments on the line or range

\end{description}
